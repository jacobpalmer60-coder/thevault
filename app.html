async function load(){ const [lg,users,rosters,players,vCsv,pCsv,traded,drafts]=await Promise.all([ fetch(`https://api.sleeper.app/v1/league/${LEAGUE_ID}`).then(r=>r.json()), fetch(`https://api.sleeper.app/v1/league/${LEAGUE_ID}/users`).then(r=>r.json()), fetch(`https://api.sleeper.app/v1/league/${LEAGUE_ID}/rosters`).then(r=>r.json()), fetch(`https://api.sleeper.app/v1/players/nfl`).then(r=>r.json()), fetch(VALUES_URL).then(r=>r.text()), fetch(PPG_URL).then(r=>r.text()), fetch(`https://api.sleeper.app/v1/league/${LEAGUE_ID}/traded_picks`).then(r=>r.json()), fetch(`https://api.sleeper.app/v1/league/${LEAGUE_ID}/drafts`).then(r=>r.json()) ]); const isSF = (lg.roster_positions||[]).includes('SUPER_FLEX'); const lastDraft= drafts.length? Math.max(...drafts.map(d=>+d.season||0)): new Date().getFullYear(); const SEASONS=[lastDraft+1,lastDraft+2,lastDraft+3], ROUNDS=[1,2,3,4]; const relevant=traded.filter(p=>SEASONS.includes(+p.season)); // VALUES - pick SF or 1QB column based on league const valMap=new Map(); Papa.parse(vCsv,{header:true,skipEmptyLines:true}).data.forEach(r=>{ if(r.sleeper_id){ const val = isSF ? (+r.val_sf_ppr_tep05 || +r.val_sf || 0) : (+r.val_1qb_ppr_tep05 || +r.val_1qb || 0); valMap.set(String(r.sleeper_id), val); } }); const ppgMap=new Map();Papa.parse(pCsv,{header:true,skipEmptyLines:true}).data.forEach(r=>{const n=r.Player||r.player;if(n)ppgMap.set(norm(n),+r['Avg_PPR_0.5_TEP']||0)}); const userMap=new Map(users.map(u=>[u.user_id,u])), rosterByUser=new Map(rosters.map(r=>[r.owner_id,r.roster_id])); const slots=(lg.roster_positions||[]).filter(s=>!['BN','IR','TAXI'].includes(s)); const optimal=list=>{const pool=[...list].sort((a,b)=>b.ppg-a.ppg);let t=0;const u=new Set();for(const s of slots){let a=[s];if(s==='FLEX')a=['RB','WR','TE'];if(s==='SUPER_FLEX')a=['QB','RB','WR','TE'];const i=pool.findIndex(p=>!u.has(p.id)&&a.includes(p.pos));if(i>=0){u.add(pool[i].id);t+=pool[i].ppg;pool.splice(i,1)}}return t}; const built=rosters.map(r=>{const u=userMap.get(r.owner_id)||{};const plist=(r.players||[]).map(pid=>{const p=players[String(pid)]||{};const n=`${p.first_name||''} ${p.last_name||''}`.trim()||p.full_name||'';return{id:String(pid),name:n,pos:p.position||'',age:p.age||0,value:valMap.get(String(pid))||0,ppg:ppgMap.get(norm(n))||0}});const tot=plist.reduce((s,p)=>s+p.value,0);const age=(()=>{const v=plist.filter(p=>p.value>0&&p.age>0);const sv=v.reduce((s,p)=>s+p.value,0);return sv?v.reduce((s,p)=>s+p.value*p.age,0)/sv:0})();return{rosterId:r.roster_id,teamName:u.metadata?.team_name||u.display_name||'Team',username:u.username||'',total:tot,qb:plist.filter(p=>p.pos==='QB').reduce((s,p)=>s+p.value,0),rb:plist.filter(p=>p.pos==='RB').reduce((s,p)=>s+p.value,0),wr:plist.filter(p=>p.pos==='WR').reduce((s,p)=>s+p.value,0),te:plist.filter(p=>p.pos==='TE').reduce((s,p)=>s+p.value,0),age,opt:optimal(plist),plist}}); const avgOpt=built.reduce((s,t)=>s+t.opt,0)/built.length;built.forEach(t=>{t.prodDelta=(t.opt/avgOpt-1)*100;t.prodPts=t.opt-avgOpt}); const own=new Map(built.map(t=>[t.rosterId,[]]));built.forEach(t=>SEASONS.forEach(s=>ROUNDS.forEach(r=>own.get(t.rosterId).push({season:s,round:r,original:t.rosterId})))); relevant.forEach(p=>{const nr=rosterByUser.get(p.owner_id);if(!nr)return;const list=own.get(p.roster_id);const i=list.findIndex(x=>x.season==p.season&&x.round==p.round);if(i>=0)list.splice(i,1);own.get(nr).push({season:+p.season,round:+p.round,original:p.roster_id})}); const tier=new Map([...built].sort((a,b)=>b.opt-a.opt).map((t,i)=>[t.rosterId,i<4?'late':i<8?'mid':'early'])); // PICKS - use SF or 1QB based on league let pickVal=new Map();try{ const csv=await fetch(PICK_VALUES_URL).then(r=>r.text()); Papa.parse(csv,{header:true,skipEmptyLines:true}).data.forEach(r=>{ if(r.season){ const tierKey=(r.tier||r.slot||'mid').toLowerCase(); const val = isSF ? (+r.sf_value||+r.value||0) : (+r.qb_value||+r.value||0); pickVal.set(`${r.season}-${r.round}-${tierKey}`,val) } }); }catch(e){console.warn('picks load failed',e)} if(!pickVal.size){SEASONS.forEach(s=>[1,2,3,4].forEach(r=>{pickVal.set(`${s}-${r}-early`,r==1?5000:r==2?2000:r==3?800:300);pickVal.set(`${s}-${r}-mid`,r==1?3500:r==2?1400:r==3?560:210);pickVal.set(`${s}-${r}-late`,r==1?2400:r==2?960:r==3?384:144)}))} built.forEach(t=>{const ps=own.get(t.rosterId)||[];let sum=0;ps.forEach(p=>{p.tier=tier.get(p.original)||'mid';p.value=pickVal.get(`${p.season}-${p.round}-${p.tier}`)||0;sum+=p.value});t.picks=ps.sort((a,b)=>a.season-b.season||a.round-b.round);t.picksValue=sum;t.capital=t.total+sum;t.cpy=t.capital/Math.max(t.age,1)}); const avgCpy=built.reduce((s,t)=>s+t.cpy,0)/built.length;built.forEach(t=>t.longDelta=(t.cpy/avgCpy-1)*100); teams=built;render();document.getElementById('status').innerHTML=`
${isSF?'SF':'1QB'} • Live • ${teams.length} teams • ${relevant.length} trades • ${SEASONS[0]}-${SEASONS[2]} picks`; }
